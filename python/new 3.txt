import boto3

def send_email_with_attachment(event, context):
    s3 = boto3.client('s3')
    ses = boto3.client('ses')

    # Get the details of the uploaded file and the bucket name
    record = event['Records'][0]
    bucket_name = record['s3']['bucket']['name']
    file_name = record['s3']['object']['key']

    # Download the file from S3
    file = s3.get_object(Bucket=bucket_name, Key=file_name)

    # Send the email with the attachment
    response = ses.send_email(
        Destination={
            'ToAddresses': [TO_EMAIL],
        },
        Message={
            'Subject': {'Data': 'Uploaded file'},
            'Body': {
                'Text': {'Data': 'A file was uploaded to S3'}
            }
        },
        Source=FROM_EMAIL,
        Attachments=[
            {
                'Name': file_name,
                'Data': file['Body'].read()
            },
        ]
    )
    return response
